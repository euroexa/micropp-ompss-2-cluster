	target_link_libraries(${testname} micropp)

	if(ENABLE_CUDA)
	  set_source_files_properties(${PROJECT_SOURCE_DIR}/test/${testfile} PROPERTIES LANGUAGE CUDA)
	endif()

endforeach ()
